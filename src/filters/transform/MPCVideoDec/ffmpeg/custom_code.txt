* libavcodec/avcodec.h
* libavcodec/h264.c
* libavcodec/h264.h
* libavcodec/mpeg12.c
* libavcodec/mpeg12.h
* libavcodec/mpegvideo.c
* libavcodec/pthread.c
* libavcodec/utils.c
* libavcodec/vc1dec.c

* libavutil/common.h
* libavutil/log.h
* libavutil/mem.c
* libavutil/mem.h
* libavutil/samplefmt.h
